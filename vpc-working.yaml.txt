# Description of what this change includes: Initial template to create VPC "VPC-dontur7609" in us-west-2 with three subnets across distinct AZs.
# Revision number: 1.0.0
# Date: 2025-10-28


AWSTemplateFormatVersion: '2010-09-09'
Description: 'Create VPC "VPC-dontur7609" in us-west-2 spanning three AZs with /20 subnets'

Resources:
  VPC:
    Type: AWS::EC2::VPC
    Properties:
      CidrBlock: 10.0.0.0/16
      EnableDnsHostnames: true
      EnableDnsSupport: true
      InstanceTenancy: default
      Tags:
        - Key: Name
          Value: VPC-dontur7609

  SubnetA:
    Type: AWS::EC2::Subnet
    Properties:
      VpcId: !Ref VPC
      CidrBlock: 10.0.0.0/20
      AvailabilityZone: !Select [0, !GetAZs 'us-west-2']
      MapPublicIpOnLaunch: false
      Tags:
        - Key: Name
          Value: VPC-dontur7609-SubnetA

  SubnetB:
    Type: AWS::EC2::Subnet
    Properties:
      VpcId: !Ref VPC
      CidrBlock: 10.0.16.0/20
      AvailabilityZone: !Select [1, !GetAZs 'us-west-2']
      MapPublicIpOnLaunch: false
      Tags:
        - Key: Name
          Value: VPC-dontur7609-SubnetB

  SubnetC:
    Type: AWS::EC2::Subnet
    Properties:
      VpcId: !Ref VPC
      CidrBlock: 10.0.32.0/20
      AvailabilityZone: !Select [2, !GetAZs 'us-west-2']
      MapPublicIpOnLaunch: false
      Tags:
        - Key: Name
          Value: VPC-dontur7609-SubnetC

Outputs:
  VpcId:
    Description: The ID of the created VPC
    Value: !Ref VPC
  SubnetAId:
    Description: Subnet A ID
    Value: !Ref SubnetA
  SubnetBId:
    Description: Subnet B ID
    Value: !Ref SubnetB
  SubnetCId:
    Description: Subnet C ID
    Value: !Ref SubnetC
